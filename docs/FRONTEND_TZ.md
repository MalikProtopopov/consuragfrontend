const API_BASE = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000';

class ApiClient {
  private async request<T>(
    endpoint: string,
    options?: RequestInit
  ): Promise<T> {
    const response = await fetch(`${API_BASE}${endpoint}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...options?.headers,
      },
      credentials: 'include', // for cookies
    });

    if (!response.ok) {
      const error = await response.json();
      throw new ApiError(error);
    }

    return response.json();
  }

  // Auth
  auth = {
    login: (data: LoginRequest) => 
      this.request<TokenResponse>('/api/v1/auth/login', {
        method: 'POST',
        body: JSON.stringify(data),
      }),
    register: (data: RegisterRequest) =>
      this.request<UserResponse>('/api/v1/auth/register', {
        method: 'POST',
        body: JSON.stringify(data),
      }),
    me: () => this.request<UserDetailResponse>('/api/v1/auth/me'),
    // ...
  };

  // Projects
  projects = {
    list: (params?: ListParams) =>
      this.request<ProjectListResponse>(`/api/v1/projects?${qs(params)}`),
    create: (data: ProjectCreate) =>
      this.request<ProjectResponse>('/api/v1/projects', {
        method: 'POST',
        body: JSON.stringify(data),
      }),
    // ...
  };

  // Avatars
  avatars = {
    list: (projectId: string, params?: ListParams) =>
      this.request<AvatarListResponse>(
        `/api/v1/projects/${projectId}/avatars?${qs(params)}`
      ),
    // ...
  };

  // Documents
  documents = {
    upload: (projectId: string, avatarId: string, file: File) => {
      const formData = new FormData();
      formData.append('file', file);
      return this.request<DocumentResponse>(
        `/api/v1/projects/${projectId}/avatars/${avatarId}/documents`,
        { method: 'POST', body: formData }
      );
    },
    // ...
  };

  // Chat
  chat = {
    createSession: (avatarId: string, source: string = 'web') =>
      this.request<ChatSessionResponse>(
        `/api/v1/chat/${avatarId}/sessions?source=${source}`,
        { method: 'POST' }
      ),
    sendMessage: (avatarId: string, sessionId: string, content: string) =>
      this.request<MessagePair>(
        `/api/v1/chat/${avatarId}/message?session_id=${sessionId}`,
        { method: 'POST', body: JSON.stringify({ content }) }
      ),
    // ...
  };
}
